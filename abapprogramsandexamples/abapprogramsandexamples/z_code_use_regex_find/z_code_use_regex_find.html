report z_code_use_regex_find.


data:
  lv_regex type string,
  lv_string type string,
  lwa_result type match_result,
  lt_result type match_result_tab.

lv_regex = '(/BIC/|/BI0/)(M|P|Q|T)\w{1,24}'.
lv_string = '/BIC/PXXX /BIC/Q1234 /BI0/MXXX'.

find all occurrences of regex lv_regex in lv_string results lt_result.


lv_regex = '(A|B|C|D)'.
lv_string = '/BIC/PXXX /BIC/Q1234 /BI0/MXXX'.

find all occurrences of regex lv_regex in lv_string results lt_result.
replace all OCCURRENCES OF regex lv_regex in lv_string with ` `.

return.
